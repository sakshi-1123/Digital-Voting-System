                <h2>Choose an Option</h2>
                <button onclick="loadVoters()">Voters list</button>
                <button onclick="showVoteForm()">Login to Vote</button>
                <button onclick="loadResults()">View Results</button>

                <!-- Voter List Section -->
                <div id="voterListSection"></div>

                <!-- Voting Form -->
                <div id="formSection" style="display:none;">
                    <h3>Vote Here</h3>
                    <input type="text" id="voterId" placeholder="Voter ID" required>
                    <input type="password" id="password" placeholder="Password" required>
                    <select id="candidateSelect" required></select>
                    <button onclick="submitVote()">Submit Vote</button>
                </div>

                <!-- Results Section -->
                <div id="resultSection"></div>
            </div>

    <script>
        // Load voters
        async function loadVoters() {
            const res = await fetch("http://localhost/127.0.0.1/voter_list/voter/voters");
            const data = await res.json();
            const section = document.getElementById("voterListSection");
            section.innerHTML = "<h3>Voters List:</h3><ul>" +
                data.map(v => `<li>ID: ${v.id}, Name: ${v.name}, Has Voted: ${v.has_voted}</li>`).join('') +
                "</ul>";
        }

        // Show vote form
        async function showVoteForm() {
            document.getElementById("formSection").style.display = "block";
            const res = await fetch("http://localhost/127.0.0.1/voter_list/voter/candidates");
            const candidates = await res.json();
            const select = document.getElementById("candidateSelect");
            select.innerHTML = `<option value="">Select a Candidate</option>` +
                candidates.map(c => `<option value="${c.id}">${c.name}</option>`).join('');
        }

        // Submit vote
        async function submitVote() {
            const voterId = document.getElementById("voterId").value;
            const password = document.getElementById("password").value;
            const candidateId = document.getElementById("candidateSelect").value;

            const res = await fetch("http://localhost/127.0.01/voter_list/voter/vote", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ voterId, password, candidateId })
            });

            const result = await res.text();
            alert(result);
        }

        // Load results
        async function loadResults() {
            const res = await fetch("http://localhost/127.0.01/voter_list/voter/results");
            const data = await res.json();
            const section = document.getElementById("resultSection");
            section.innerHTML = "<h3>Election Results:</h3><ul>" +
                data.map(r => `<li>${r.name}: ${r.votes} votes</li>`).join('') +
                "</ul>";
        }
    </script>
